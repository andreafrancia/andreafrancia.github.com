Automate user creation

sudo /usr/sbin/useradd sysop
echo sysop: | sudo /usr/sbin/chpasswd
sudo mkdir -p /home/sysop
sudo chown sysop /home/sysop
sudo /usr/sbin/usermod -G admin sysop
